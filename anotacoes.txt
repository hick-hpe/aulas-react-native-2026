3 - babel.config.js (Babel, transpilador de JavaScript)
module.exports = function (api) {
  api.cache(true);
  return {
    presets: ['babel-preset-expo'],
  };
};

// transpilador
analisar ("ler e entender") código moderno e reescrevê-lo usando construções de sintaxe mais antigas

npx expo start --clear (ou -c)

// bundler
